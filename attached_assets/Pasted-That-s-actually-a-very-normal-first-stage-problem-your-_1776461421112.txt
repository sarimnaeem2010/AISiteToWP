function detectLayout(element) {

function generateId() {

function cleanDOM(root) {
  root.querySelectorAll("script, style, noscript").forEach(el => el.remove());
  return root;

// ------------------------------
// LAYOUT DETECTION
// ------------------------------
function detectLayout(element) {
  const style = window.getComputedStyle(element);

  return {
    display: style.display,
    isFlex: style.display.includes("flex"),
    isGrid: style.display.includes("grid"),
    direction: style.flexDirection || "row",
    gap: style.gap || null
  };
}

// ------------------------------
// CLASS-BASED FALLBACK
// ------------------------------
function hasLayoutClass(el) {
  const cls = el.className || "";
  return (
    cls.includes("flex") ||
    cls.includes("grid") ||
    cls.includes("row") ||
    cls.includes("col")
  );
}

// ------------------------------
// SIMILAR CHILD DETECTION
// ------------------------------
function areSiblingsSimilar(element) {
  const children = Array.from(element.children);
  if (children.length < 2) return false;

  const firstTag = children[0].tagName;
  return children.every(child => child.tagName === firstTag);
}

// ------------------------------
// SECTION DETECTION
// ------------------------------
function isSection(element) {
  const tag = element.tagName.toLowerCase();

  if (["section", "header", "footer", "main"].includes(tag)) return true;
  if (element.children.length >= 2) return true;

  return false;
}

// ------------------------------
// WIDGET DETECTION
// ------------------------------
function detectWidgetType(el) {
  const tag = el.tagName.toLowerCase();
  const cls = el.className || "";

  if (["h1","h2","h3","h4","h5","h6"].includes(tag)) return "heading";
  if (tag === "p" || tag === "span") return "text-editor";
  if (tag === "img") return "image";

  if (
    tag === "button" ||
    (tag === "a" && cls.includes("btn"))
  ) return "button";

  if (tag === "ul" || tag === "ol") return "text-editor";

  return "text-editor"; // fallback
}

// ------------------------------
// EXTRACT SETTINGS
// ------------------------------
function extractWidgetSettings(el, type) {
  if (type === "heading") {
    return {
      title: el.innerText.trim(),
      tag: el.tagName.toLowerCase()
    };
  }

  if (type === "text-editor") {
    return {
      editor: el.innerHTML
    };
  }

  if (type === "image") {
    return {
      image: {
        url: el.src
      }
    };
  }

  if (type === "button") {
    return {
      text: el.innerText.trim(),
      link: {
        url: el.href || "#"
      }
    };
  }

  return {};
}

// ------------------------------
// CREATE WIDGET
// ------------------------------
function createWidget(el) {
  const type = detectWidgetType(el);

  return {
    id: generateId(),
    elType: "widget",
    widgetType: type,
    settings: extractWidgetSettings(el, type)
  };
}

// ------------------------------
// DETECT WIDGETS INSIDE COLUMN
// ------------------------------
function detectWidgets(element) {
  const widgets = [];

  element.childNodes.forEach(node => {
    if (node.nodeType !== 1) return;

    // skip wrappers with single child
    if (node.children.length === 1 && !node.innerText.trim()) return;

    widgets.push(createWidget(node));
  });

  return widgets;
}

// ------------------------------
// CREATE COLUMN
// ------------------------------
function createColumn(element, totalColumns = 1) {
  return {
    id: generateId(),
    elType: "column",
    settings: {
      _column_size: 100 / totalColumns
    },
    elements: detectWidgets(element)
  };
}

// ------------------------------
// DETECT COLUMNS
// ------------------------------
function detectColumns(element) {
  const layout = detectLayout(element);
  const children = Array.from(element.children);

  // FLEX ROW
  if (layout.isFlex && layout.direction === "row") {
    return children.map(child =>
      createColumn(child, children.length)
    );
  }

  // GRID
  if (layout.isGrid) {
    return children.map(child =>
      createColumn(child, children.length)
    );
  }

  // CLASS FALLBACK
  if (hasLayoutClass(element)) {
    return children.map(child =>
      createColumn(child, children.length)
    );
  }

  // SIMILAR STRUCTURE
  if (areSiblingsSimilar(element)) {
    return children.map(child =>
      createColumn(child, children.length)
    );
  }

  // DEFAULT SINGLE COLUMN
  return [createColumn(element, 1)];
}

// ------------------------------
// CREATE SECTION
// ------------------------------
function createSection(element) {
  return {
    id: generateId(),
    elType: "section",
    settings: {},
    elements: detectColumns(element)
  };
}

// ------------------------------
// MAIN TREE BUILDER
// ------------------------------
function buildElementorJSON(htmlString) {
  const parser = new DOMParser();
  let doc = parser.parseFromString(htmlString, "text/html");

  const body = cleanDOM(doc.body);

  const sections = [];

  Array.from(body.children).forEach(el => {
    if (isSection(el)) {
      sections.push(createSection(el));
    }
  });

  return sections;
}
